instalaciones para empezar backend

-npm init                   "creamos proyecto de node.js"
-npm i --save express       "libreria de express"
-npm i --save dotenv        "Para usar las variables de entorno"
-npm i --save-dev nodemon   "para al guardar los cambios nos actualize el backend"

-creamos los scripts
"start": "nodemon index.js",

-creamos la carpeta .env    van a estar alojadas las variables de entorno
-creamos carpeta "routes" con su archivo index.js      va a contener todas las rutas de los enpoint
-creamos la carpeta middlewares    va a contener todos los middlewares. funciones que se ejecutan
                                   antes de que se ejecuten las peticiones

-creamos la carpeta controllers    va a contener todos los controladores de cada una de las peticiones


-exportamo todo lo neceario en index.js

const dotenv = require('dotenv') 
const express = require('express')  

const routes = require('./routes')

dotenv.config()

const app = express()

app.use(express.json())
